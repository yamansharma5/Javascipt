Spread and Rest in JavaScript

Spread Operator (...)

Use Case: To expand (unpack) elements of an iterable (like array, string, or object).

In Arrays

let arr1 = [1, 2, 3];
let arr2 = [...arr1, 4, 5];
console.log(arr2); // [1, 2, 3, 4, 5]

In Objects

let obj1 = { a: 1, b: 2 };
let obj2 = { ...obj1, c: 3 };
console.log(obj2); // { a: 1, b: 2, c: 3 }

In Function Calls

function sum(a, b, c) {
  return a + b + c;
}
let nums = [1, 2, 3];
console.log(sum(...nums)); // 6

Summary: Spread expands an array/object/string into individual elements.

Rest Operator (...)

Use Case: To collect multiple elements into a single array (or object).

Function Parameters

function sum(...nums) {
  return nums.reduce((a, b) => a + b, 0);
}
console.log(sum(1, 2, 3, 4)); // 10

Destructuring Arrays

let [first, ...rest] = [1, 2, 3, 4];
console.log(first); // 1
console.log(rest);  // [2, 3, 4]

Destructuring Objects

let { a, ...others } = { a: 1, b: 2, c: 3 };
console.log(a);      // 1
console.log(others); // { b: 2, c: 3 }

Summary: Rest collects remaining values into an array or object.

Visual Example

function example(a, b, ...rest) {
  console.log(a);     // 1
  console.log(b);     // 2
  console.log(rest);  // [3, 4, 5]
}

let values = [1, 2, 3, 4, 5];
example(...values);
